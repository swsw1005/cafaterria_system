CREATE TABLE MENU(
    orderdate date PRIMARY KEY,
    menu1 VARCHAR2(30),
    menu2 VARCHAR2(30),
    menu3 VARCHAR2(30),
    menu4 VARCHAR2(30),
    menu5 VARCHAR2(30)
);

CREATE TABLE ORDERLIST(
    no number PRIMARY KEY,
    orderdate date ,
    count NUMBER DEFAULT 0,
    name VARCHAR2(30)
);

    no number PRIMARY KEY,
    id varchar2(30) not null,
    pw varchar2(30) not null,
    state varchar2(6) NOT NULL CHECK(state in('활동','중지'))
);
